def standardize(df):
    std = df.std(ddof=0)
    mean = df.mean(axis='index')
    return (df.sub(mean,axis='columns')).div(std,axis='columns')
In [4]:
def correlation(x, y):
    std_x = (x - x.mean())/x.std(ddof=0)
    std_y = (y - y.mean())/y.std(ddof=0)
    return (std_x*std_y).mean()
